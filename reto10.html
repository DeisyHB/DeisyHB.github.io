Reto #10

Desarrollar un algoritmo que reciba un número entero y lo convierta a binario y hexadecimal.

Solución


#1 función para convertir número entero a binario
numero_entero=int(input("Ingrese el número que desea convertir a número binario: "))

def entero_binario(numero_entero):
numero_binario = 0
multiplicador = 1

while numero_entero != 0:

numero_binario = numero_binario + numero_entero % 2 * multiplicador
numero_entero //= 2
multiplicador *= 10

return numero_binario

#Se imprime el número binario

print(entero_binario(numero_entero))

#2 Algortimo que convierte números enteros en hexadecimales.

def entero_hexadecimal():
entero = int(input('Introduce el número que deseas convertir en hexadecimal:\n'))
convertido = hex(entero)[2:]
print(convertido)

entero_hexadecimal()
