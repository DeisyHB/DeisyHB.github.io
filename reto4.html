Reto #4

Desarrollar un algoritmo que simule la descomposición numérica. Recibirá un número de 1 a n y deberá mostrar los factores y los cocientes.

Solución

def descomposicion(num):
divisores = []
cocientes = [num]

num_divisor = 2
while num_divisor <= num:
if num % num_divisor == 0:
divisores.append(num_divisor)
cocientes.append(num // num_divisor)
num = num // num_divisor
else:
num_divisor += 1

print("60")
print('Los factores son ', divisores);
print('Los cocientes son ', cocientes);

descomposicion(60);
