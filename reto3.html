Reto #3 Cifrado César


Parte 1:

Desarrollar un algoritmo que reciba una letra del Abecedario , la cual será la base para codificar y decodificar mensajes.
Esta letra se convertirá en la primera de la secuencia y determinará el cambio de cada letra siguiente, por ejemplo:

Ingresa la B, todo el abecedario cambia de la siguiente forma:
A = B
B = C
C = D
……..
Z = A

Parte 2:

El algoritmo debe recibir una palabra cifrada, y la base original y deberá decodificarla, mostrando la palabra real.

Solución

#Reto Cifrado César
# 26 = Letras del alfabeto
alfabeto = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'

def encript(message, letter_base):
letter_base = letter_base.upper()
desplazamiento = alfabeto.index(letter_base)

# Concat
cifrado = ''

for letra in message:
if letra.upper() in alfabeto:
nueva_letra = alfabeto[(alfabeto.index(letra.upper()) + desplazamiento) % 26]

if letra.islower():
nueva_letra = nueva_letra.lower()

cifrado += nueva_letra
else:
cifrado += letra
return cifrado
def decrypt(message, letra_base):
letra_base = letra_base.upper()
desplazamiento = alfabeto.index(letra_base)

# Concat
descifrado = ''
for letra in message:
if letra.upper() in alfabeto:
nueva_letra = alfabeto[(alfabeto.index(letra.upper()) - desplazamiento) % 26]

if letra.islower():
nueva_letra = nueva_letra.lower()

descifrado += nueva_letra
else:
descifrado += letra
return descifrado

mensaje = 'Hi, It\'s me. :)'
letter_base = 'S'
mensaje_cifrado = encript(mensaje, letter_base)
mensaje_descifrado = decrypt(mensaje_cifrado, letter_base)

print('Mensaje original: ' + mensaje)
print('Mensaje cifrado: ' + mensaje_cifrado)
print('Mensaje descifrado: ' + mensaje_descifrado)
