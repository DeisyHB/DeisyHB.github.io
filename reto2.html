Reto #2

Escribir un algoritmo que reciba una x cantidad de números y los devuelva organizados ascendente y descendentemente. En caso de existir números repetidos sólo deberá mostrar uno.
Ejemplo: 1 9 5 7 9 3 1
Salida:

1 3 5 7 9
9 7 5 3 1

Solución

#Se crea una variable para ingresar los números de la lista.

cantidad_num=int(input("¿Cuántos números vas a ingresar?"))
my_list=[]
for i in range(cantidad_num):
numeros=int(input(f"Ingresa el número {i+1}:"))
my_list.append(numeros)

#los números que digitaste fueron
print("Te mostraré los números que ingresaste: ")
for numeros in my_list:
print(numeros)

# esto imprime la lista NO ordenada
print("Lista desordenada: ", my_list)

#Eliminemos valores duplicados. Una forma es pasarlo a conjunto, ya que los conjuntos no tienen elementos repetidos
num_unicos=list(set(my_list))
# Ordenar la lista
num_unicos.sort()

# lista orden ascendente
print("Lista orden ascendente: ", num_unicos)

# lista orden descendente
orden_desc = sorted(num_unicos, reverse=True)
print("orden descendente", orden_desc)
